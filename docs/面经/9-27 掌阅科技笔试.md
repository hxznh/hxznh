public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        String[] input = in.next().split(",");
        int k = in.nextInt();
        int[] nums = new int[input.length];
        for (int i = 0; i < input.length; i++) {
            nums[i] = Integer.parseInt(input[i]);
        }
        Map<Integer, Integer> map = new HashMap<>();
        List<Pair> list = new ArrayList<>();
        for (int x: nums) {
            if (!map.containsKey(x)) {
                map.put(x, list.size());
                list.add(new Pair(x, 1));
            } else {
                int idx = map.get(x);
                list.get(idx).count++;
            }
        }
        list.sort((o1, o2) -> o2.count - o1.count);

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < k - 1; i++) {
            sb.append(list.get(i).value + ",");
        }
        sb.append("" + list.get(k - 1).value);
        System.out.println(sb.toString());
    }
}

class Pair {
    int value;
    int count;
    Pair() {}
    Pair(int _value, int _count) {
        count = _count;
        value = _value;
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        String[] input = in.nextLine().split(", ");
        int[] numA = new int[input.length];
        for (int i = 0; i < input.length; i++) {
            numA[i] = Integer.parseInt(input[i]);
        }
        input = in.nextLine().split(", ");
        int[] numB = new int[input.length];
        for (int i = 0; i < input.length; i++) {
            numB[i] = Integer.parseInt(input[i]);
        }

        Arrays.sort(numA);
        Arrays.sort(numB);
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < numB.length; i++) {
            int x = numB[i];
            int idx = binarySearch(numA, x);
            res = Math.min(Math.abs(numA[idx] - x), res);
            if (idx + 1 < numA.length) {
                res = Math.min(Math.abs(numA[idx + 1] - x), res);
            }
        }

        System.out.println(res);
    }

    public static int binarySearch(int[] nums, int x) {
        int l = 0, r = nums.length - 1;
        while (l < r) {
            int mid = l + (r - l + 1) / 2;
            if (nums[mid] >= x) {
                r = mid - 1;
            } else {
                l = mid;
            }
        }

        return l;
    }





}
